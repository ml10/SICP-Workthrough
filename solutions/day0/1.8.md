;; Implement cube root

(defn cube [x] (Math/pow x 3))

(defn good-enough? [guess x]
	(< (Math/abs (- (cube guess) x)) 0.001))

(defn improve [guess x]
	(/ (+ (/ x (* guess guess))
	   (* 2 guess))
	3))

(defn cbrt-iter [guess x]
	(if (good-enough? guess x)
	 guess
	 (cbrt-iter (improve guess x) x)))

(defn cbrt [x] (cbrt-iter 1.0 x)))
