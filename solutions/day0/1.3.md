;;Exercise 1.3. Define a procedure that takes three numbers as arguments and returns the sum of the squares of the two larger numbers.

;;square
(defn square [x] (* x x))

;;sum-of-squares
(defn sum-of-squares [x y] (+ (square x) (square y)))

;;Possible Solution 1:
(defn sum-of-largest-square [x y z]
	(- (apply + (map square ([x y z])))
		 (square (min x y z))))

;;Possible Solution 2:
(defn sum-of-largest-square [x y z]
  (let [top2 (take 2 (reverse (sort [x y z])))]
    (apply sum-of-squares top2)))
