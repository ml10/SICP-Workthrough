(defn good-enough? [guess x]
   (< (Math/abs (/ (- (improve guess x) guess)
              guess))
      0.001))

;; below same as provided in book
(defn sqrt-iter [guess x]  
  (if (good-enough? guess x)  
      guess  
      (sqrt-iter (improve guess x)  
                 x))) 

(defn improve [guess x]  
  (average guess (/ x guess)))

(defn average [x y]  
  (/ (+ x y) 2)) 

(defn sqrt [x]  
  (sqrt-iter 1.0 x))
